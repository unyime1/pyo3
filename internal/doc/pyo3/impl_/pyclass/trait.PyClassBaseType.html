impl<T: PyClass> PyClassBaseType for T

All mutable PyClasses can be used as a base type.

In the future this will be extended to immutable PyClasses too.

type LayoutAsBase = PyCell<T>

type BaseNativeType = <T as PyClassImpl>::BaseNativeType

type ThreadChecker = <T as PyClassImpl>::ThreadChecker

type Initializer = PyClassInitializer<T>

type PyClassMutability = <T as PyClassImpl>::PyClassMutability
